function gf(r,keys){
  for(var i=0;i<keys.length;i++){
    if(r[keys[i]]!==undefined&&r[keys[i]]!==null&&String(r[keys[i]]).trim()!=='') return String(r[keys[i]]).trim();
  }
  return '';
}
function sn(v){var n=parseFloat(v);return isNaN(n)?0:n;}
function fmtDate(d){return d?new Date(d).toLocaleDateString('ar-EG'):'—';}

// ══ WAREHOUSE STOCK HELPERS ═══════════════════════════════
// Get live warehouse stock for a given code/barcode from warehouseRows
function getWarehouseStock(code, barcode){
  var row = warehouseRows.find(function(r){
    return gf(r,['كود الصنف','code'])===String(code).trim()
      || (barcode && barcode!=='—' && gf(r,['باركود','barcode'])===String(barcode).trim());
  });
  return row ? sn(row['رصيد المخزن']||row['warehouse']||0) : null;
}

// Deduct warehouse stock by qty for a given code/barcode; returns actual deducted
function deductWarehouseStock(code, barcode, qty){
  var row = warehouseRows.find(function(r){
    return gf(r,['كود الصنف','code'])===String(code).trim()
      || (barcode && barcode!=='—' && gf(r,['باركود','barcode'])===String(barcode).trim());
  });
  if(!row) return 0;
  var cur = sn(row['رصيد المخزن']||row['warehouse']||0);
  var deduct = Math.min(qty, Math.max(0, cur));
  row['رصيد المخزن'] = cur - deduct;
  if(row['warehouse'] !== undefined) row['warehouse'] = row['رصيد المخزن'];
  // Invalidate cache for all branch rows with this code
  masterBranchRows.forEach(function(mr, idx){
    var mCode = gf(mr,['كود الصنف','code']);
    var mBc = gf(mr,['باركود','barcode']);
    if(mCode===String(code).trim() || (barcode && mBc===String(barcode).trim())){
      mr['رصيد المخزن'] = row['رصيد المخزن'];
      if(mr['warehouse']!==undefined) mr['warehouse'] = row['رصيد المخزن'];
      // Invalidate cache
      calcCache.forEach(function(v,k){if(k.startsWith(idx+'|')) calcCache.delete(k);});
    }
  });
  return deduct;
}

// Total warehouse stock across all items
function getTotalWarehouseStock(){
  return warehouseRows.reduce(function(s,r){return s+sn(r['رصيد المخزن']||r['warehouse']||0);},0);
}

function genBranchTemplate(){
  if(!warehouseRows.length){alert('⚠️ يرجى رفع ملف المخزن أولاً.');goPage('warehouse');return;}
  var aliases={'التقسيم السلعي':['التقسيم السلعي','division'],'القسم':['القسم','department'],'المجموعة الرئيسية':['المجموعة الرئيسية','main_group'],'المجموعة الفرعية':['المجموعة الفرعية','sub_group'],'اسم المورد':['اسم المورد','supplier'],'كود الصنف':['كود الصنف','code'],'اسم الصنف':['اسم الصنف','item_name','name'],'باركود':['باركود','barcode'],'رصيد المخزن':['رصيد المخزن','warehouse'],'رشيو المنتج':['رشيو المنتج','ratio']};
  showLoading('جارٍ توليد الشيت المجمع…','');
  var rows=[BR_ALL_COLS], brIdx=0;
  function processBatch(){
    var end=Math.min(brIdx+3,BRANCHES.length);
    for(var bi=brIdx;bi<end;bi++){
      var br=BRANCHES[bi];
      warehouseRows.forEach(function(mr){
        rows.push(BR_ALL_COLS.map(function(col){
          if(col==='الفرع') return br;
          if(BR_INPUT_COLS.indexOf(col)>-1) return '';
          return gf(mr,aliases[col]||[col]);
        }));
      });
    }
    brIdx=end;
    updateLoadBar(Math.round(brIdx/BRANCHES.length*90),'الفروع: '+brIdx+'/'+BRANCHES.length);
    if(brIdx<BRANCHES.length){setTimeout(processBatch,10);}
    else{
      updateLoadBar(95,'كتابة الملف…');
      setTimeout(function(){
        try{
          var ws=XLSX.utils.aoa_to_sheet(rows);
          ws['!cols']=BR_ALL_COLS.map(function(h){return{wch:Math.max(h.length*1.8,14)};});
          var wb2=XLSX.utils.book_new();
          XLSX.utils.book_append_sheet(wb2,ws,'بيانات الفروع المجمعة');
          XLSX.writeFile(wb2,'تيمبلت_الفروع_Tree.xlsx');
          updateLoadBar(100,'تم!');
          setTimeout(hideLoading,400);
        }catch(err){hideLoading();alert('❌ خطأ في إنشاء الشيت:\n'+err.message);}
      },50);
    }
  }
  setTimeout(processBatch,50);
}

function parseBranchFile(file,period){
  showLoading('جارٍ قراءة ملف الفروع المجمع…','يتم تحميل الملف');
  _perfStart=Date.now();
  var reader=new FileReader();
  reader.onload=function(ev){
    updateLoadBar(15,'جارٍ تحليل الشيت…');
    setTimeout(function(){
      try{
        var wb=XLSX.read(ev.target.result,{type:'array',cellDates:false,sheetStubs:false});
        updateLoadBar(50,'جارٍ تحويل البيانات…');
        setTimeout(function(){
          try{
            var ws=wb.Sheets[wb.SheetNames[0]];
            var rawJson=XLSX.utils.sheet_to_json(ws,{defval:'',raw:true});
            var total=rawJson.length;
            if(!total){hideLoading();alert('⚠️ الملف فارغ أو لا يحتوي بيانات.');return;}
            updateLoadBar(68,'معالجة '+total+' سجل…');
            masterBranchRows=[];calcCache.clear();
            var CHUNK=25000, idx=0;
            function processChunk(){
              try{
                var end=Math.min(idx+CHUNK,total);
                for(var i=idx;i<end;i++){
                  var r=rawJson[i];
                  var f=r['الفرع']||r['branch']||r['الالفرع']||'';
                  var co=r['كود الصنف']||r['code']||'';
                  if(String(f).trim()!==''&&String(co).trim()!==''&&String(co).indexOf('لا تعدّل')===-1){
                    r._branch=String(f).trim();
                    r._days=period.salDays;
                    r._period=period;
                    masterBranchRows.push(r);
                  }
                }
                idx=end;
                updateLoadBar(68+Math.round((idx/total)*27),'معالجة '+idx+' / '+total);
                if(idx<total){setTimeout(processChunk,1);}
                else{
                  var elapsed=Date.now()-_perfStart;
                  updateLoadBar(97,'تحديث الواجهة…');
                  setTimeout(function(){
                    try{
                      renderBranchUploadList();
                      refreshDataRepresentation();
                      updateAnalysisPage();
                      applyTransitFiltersAndRender();
                      updateOrderPage();
                      updateLoadBar(100,'اكتمل في '+elapsed+'ms!');
                      setTimeout(function(){
                        hideLoading();goPage('dashboard');
                        var uq=[...new Set(masterBranchRows.map(function(r){return r._branch;}))];
                        alert('✅ تم رفع الملف!\n'+masterBranchRows.length+' سجل | '+uq.length+' فرع | '+elapsed+'ms');
                      },200);
                    }catch(err){hideLoading();alert('❌ خطأ في تحديث الواجهة:\n'+err.message);}
                  },30);
                }
              }catch(err){hideLoading();alert('❌ خطأ في معالجة البيانات:\n'+err.message);}
            }
            setTimeout(processChunk,10);
          }catch(err){hideLoading();alert('❌ خطأ في تحليل الشيت:\n'+err.message);}
        },10);
      }catch(err){hideLoading();alert('❌ خطأ في قراءة الملف:\n'+err.message);}
    },10);
  };
  reader.onerror=function(){hideLoading();alert('❌ تعذّر قراءة الملف.');}; 
  reader.readAsArrayBuffer(file);
}

function calcRow(r,idx){
  var pd=sn(r._days)||30;
  var cl=sn(r['رصيد آخر المدة'])||sn(r['closing']);
  var sa=sn(r['مبيعات الفترة'])||sn(r['sales']);
  var cacheKey=(idx!==undefined?idx:0)+'|'+cl+'|'+sa+'|'+cfg.s_break+'|'+cfg.s_over+'|'+cfg.cov_max;
  if(calcCache.has(cacheKey)) return calcCache.get(cacheKey);

  var op=sn(r['رصيد أول المدة'])||sn(r['opening']);
  var wh=sn(r['رصيد المخزن'])||sn(r['warehouse']);
  var rt=sn(r['رشيو المنتج'])||sn(r['ratio'])||1;

  var avg=pd>0?sa/pd:0;
  var covRaw=avg>0?cl/avg:9999;
  var rp=rt>0?(cl/rt)*100:0;

  var salesRatio;
  if(sa===0){salesRatio=0;}
  else if(cl===0){salesRatio=999999;}
  else{salesRatio=sa/cl;}

  var ti;
  if(sa===0){ti=0;}
  else if(cl===0){ti=999999;}
  else{
    var cov_for_ti=avg>0?cl/avg:999999;
    ti=cov_for_ti>0?(avg*salesRatio)/cov_for_ti:0;
  }

  var avgStock=(op+cl)/2;
  var turnover=avgStock>0?sa/avgStock:0;

  var invStatus;
  if     (rp<0)           invStatus='خطا كارته';
  else if(rp===0)         invStatus='غير متوفر';
  else if(rp<cfg.s_break) invStatus='كسر';
  else if(rp<=cfg.s_over) invStatus='متوفر';
  else                    invStatus='اوفر استوك';

  var hasSales=sa>0?'يوجد بيع':'لا يوجد بيع';
  var rec,rc;
  if(op===0&&cl===0&&sa===0){rec='منتج غير متاح في الفرع';rc='bgr';}
  else if(hasSales==='يوجد بيع'&&(invStatus==='كسر'||invStatus==='غير متوفر')){rec='اطلب الان';rc='bg';}
  else if(hasSales==='لا يوجد بيع'&&invStatus==='كسر'){rec='مراجعة الصنف';rc='ba';}
  else if((hasSales==='لا يوجد بيع'&&(invStatus==='اوفر استوك'||invStatus==='متوفر'))||(invStatus==='اوفر استوك'&&covRaw>=cfg.cov_max)){rec='قرارات تحسين';rc='bpu';}
  else if(hasSales==='يوجد بيع'&&(invStatus==='متوفر'||invStatus==='اوفر استوك')){rec='متوسط';rc='bt';}
  else{rec='متوسط';rc='bt';}

  var result=Object.assign({},r,{
    _op:op,_cl:cl,_sa:sa,_wh:wh,_rt:rt,
    _avg:avg,_covRaw:covRaw,
    _cov:covRaw>=9999?'—':parseFloat(covRaw.toFixed(5)),
    _rp:rp,
    _salesRatio:salesRatio,
    _transferIndex:ti,
    _turnover:turnover,
    _rec:rec,_rc:rc,
    _invStatus:invStatus,_hasSales:hasSales
  });

  calcCache.set(cacheKey,result);
  return result;
}

// ══ REFRESH DATA ══════════════════════════════════════════
var _kpiCache={on:0,mt:0,im:0,rv:0,n:0};

function refreshDataRepresentation(){
  var sel=document.getElementById('brSel');
  var selectedBranch='__all__';
  if(user&&user.role==='branch') selectedBranch=user.branch;
  else selectedBranch=sel?sel.value:'__all__';

  filteredRows=[];
  var on=0,mt=0,im=0,rv=0;
  for(var i=0;i<masterBranchRows.length;i++){
    var r=masterBranchRows[i];
    if(selectedBranch!=='__all__'&&r._branch!==selectedBranch) continue;
    var cr=calcRow(r,i);
    filteredRows.push(cr);
    if(cr._rec==='اطلب الان') on++;
    else if(cr._rec==='متوسط') mt++;
    else if(cr._rec==='قرارات تحسين') im++;
    else if(cr._rec==='مراجعة الصنف') rv++;
  }
  _kpiCache={on:on,mt:mt,im:im,rv:rv,n:filteredRows.length};
  updKpis(); updDash(); updAlerts(); updDonut(); updateBranchStatus();
  applyFiltersAndRender();
}

function updDonut(){
  var c=_kpiCache,n=c.n||1,C=251,off=0;
  var a1=Math.round(c.on/n*C),a2=Math.round(c.mt/n*C),a3=Math.round(c.im/n*C),a4=Math.max(C-a1-a2-a3,0);
  [['d1',a1],['d2',a2],['d3',a3],['d4',a4]].forEach(function(x){
    var el=document.getElementById(x[0]);if(!el)return;
    el.setAttribute('stroke-dasharray',x[1]+' '+(C-x[1]));
    el.setAttribute('transform','rotate('+((-90)+off*360/C)+' 55 55)');
    off+=x[1];
  });
}

  var dvs=[...new Set(filteredRows.map(function(r){return gf(r,['التقسيم السلعي']);}))].filter(Boolean);
  var dps=[...new Set(filteredRows.map(function(r){return gf(r,['القسم']);}))].filter(Boolean);

function applyFiltersAndRender(){
  var t0=Date.now();
  var rf=document.getElementById('fRec')?document.getElementById('fRec').value:'all';
  var fd=document.getElementById('fDiv')?document.getElementById('fDiv').value:'all';
  var fp=document.getElementById('fDpt')?document.getElementById('fDpt').value:'all';
  var sr=(document.getElementById('srch')?document.getElementById('srch').value.trim():'').toLowerCase();
  var fl=filteredRows;
  if(rf!=='all') fl=fl.filter(function(r){return r._rec===rf;});
  if(fd!=='all') fl=fl.filter(function(r){return gf(r,['التقسيم السلعي'])===fd;});
  if(fp!=='all') fl=fl.filter(function(r){return gf(r,['القسم'])===fp;});
  if(sr) fl=fl.filter(function(r){
    return [gf(r,['كود الصنف']),gf(r,['اسم الصنف']),gf(r,['باركود'])].some(function(v){return String(v).toLowerCase().indexOf(sr)>-1;});
  });
  pagedRows=fl;
  curPage=1;
  totalPages=Math.max(1,Math.ceil(fl.length/pageSize));
  document.getElementById('cntLbl').textContent='إجمالي: '+fl.length+' صنف';
  document.getElementById('ps1').textContent=masterBranchRows.length.toLocaleString();
  document.getElementById('ps2').textContent=fl.length.toLocaleString();
  document.getElementById('ps4').textContent=(Date.now()-t0)+'ms';
  renderPage();
}

function renderPage(){

  var slice=pagedRows.slice(start,end);
  document.getElementById('ps3').textContent=slice.length.toLocaleString();
  var invMap={'خطا كارته':'br','غير متوفر':'bgr','كسر':'ba','متوفر':'bg','اوفر استوك':'bpu'};
  var tb=document.getElementById('dTb');
  if(!slice.length){
    tb.innerHTML='<tr><td colspan="24" style="text-align:center;padding:2rem;color:var(--tx3)">لا توجد نتائج مطابقة</td></tr>';
    renderPagination();
    document.getElementById('pgInfo').textContent='صفحة '+curPage+' من '+totalPages;
    return;
  }
  var html='';
  for(var i=0;i<slice.length;i++){
    var r=slice[i];
    var gn=start+i+1;
    var ib=invMap[r._invStatus]||'bgr';
    var sb=r._hasSales==='يوجد بيع'?'bt':'bgr';
    var srDisp=r._salesRatio>=999990?'∞':r._salesRatio.toFixed(5);
    var tiDisp=r._transferIndex>=999990?'∞':r._transferIndex.toFixed(5);

    html+='<tr><td>'+gn+'</td>'
      +'<td style="font-weight:700;color:var(--ac)">'+r._branch+'</td>'
      +'<td>'+gf(r,['التقسيم السلعي'])+'</td><td>'+gf(r,['القسم'])+'</td>'
      +'<td>'+gf(r,['المجموعة الرئيسية'])+'</td><td>'+gf(r,['المجموعة الفرعية'])+'</td>'
      +'<td>'+gf(r,['اسم المورد'])+'</td>'
      +'<td style="font-weight:600">'+gf(r,['كود الصنف'])+'</td>'
      +'<td>'+gf(r,['اسم الصنف'])+'</td>'
      +'<td style="color:var(--tx3);font-size:11px">'+gf(r,['باركود'])+'</td>'
      +'<td>'+Math.round(r._op)+'</td>'
      +'<td>'+Math.round(r._cl)+'</td>'
      +'<td style="font-weight:700">'+Math.round(r._sa)+'</td>'
      +'<td>'+r._avg.toFixed(5)+'</td>'
      +'<td style="font-weight:600;color:var(--bl2)">'+srDisp+'</td>'
      +'<td style="font-weight:600;color:var(--tl2)">'+r._turnover.toFixed(5)+'</td>'
      +'<td style="font-weight:700">'+r._cov+'</td>'
      +'<td>'+Math.round(r._wh)+'</td>'
      +'<td>'+Math.round(r._rt)+'</td>'
      +'<td style="font-weight:700;color:var(--ac2)">'+r._rp.toFixed(5)+'%</td>'
      +'<td class="'+tiClass+'">'+tiDisp+'</td>'
      +'<td><span class="badge '+ib+'">'+r._invStatus+'</span></td>'
      +'<td><span class="badge '+sb+'">'+r._hasSales+'</span></td>'
      +'<td><span class="badge '+r._rc+'">'+r._rec+'</span></td></tr>';
  }
  tb.innerHTML=html;
  renderPagination();
  document.getElementById('pgInfo').textContent='صفحة '+curPage+' من '+totalPages+' | '+pagedRows.length.toLocaleString()+' نتيجة';
}

function renderPagination(){
  var pg=document.getElementById('pgNums'),html='';
  var range=7,half=Math.floor(range/2);
  var start=Math.max(1,curPage-half);
  var end=Math.min(totalPages,start+range-1);
  if(end-start<range-1) start=Math.max(1,end-range+1);
  if(start>1){html+='<button class="pg-btn" onclick="goPage2(1)">1</button>';if(start>2) html+='<span style="color:var(--tx3);padding:0 4px">…</span>';}
  for(var i=start;i<=end;i++) html+='<button class="pg-btn'+(i===curPage?' active':'')+'" onclick="goPage2('+i+')">'+i+'</button>';
  if(end<totalPages){if(end<totalPages-1) html+='<span style="color:var(--tx3);padding:0 4px">…</span>';html+='<button class="pg-btn" onclick="goPage2('+totalPages+')">'+totalPages+'</button>';}
  pg.innerHTML=html;
  document.getElementById('pgFirst').disabled=(curPage===1);
  document.getElementById('pgPrev').disabled=(curPage===1);
  document.getElementById('pgNext').disabled=(curPage===totalPages);
  document.getElementById('pgLast').disabled=(curPage===totalPages);
}

function goPage2(n){
  if(n<1||n>totalPages) return;
  curPage=n; renderPage();
  document.getElementById('main').scrollTop=0;
}

function getOrderRows(branchName){
  var rows=[];
  for(var i=0;i<masterBranchRows.length;i++){
    var r=masterBranchRows[i];
    if(r._branch!==branchName) continue;
    var cr=calcRow(r,i);
    if(ORDER_DECISIONS.indexOf(cr._rec)>-1) rows.push(cr);
  }
  return rows;
}

function updateOrderPage(){
  if(!user||user.role!=='branch') return;
  var br=user.branch;
  document.getElementById('orderBranchLabel').textContent='الفرع: '+br+' | فترة التحليل: '+globalPeriod.salDays+' يوماً';
  var rows=getOrderRows(br);
  if(!rows.length){
    document.getElementById('orderNoDataMsg').style.display='block';
    document.getElementById('orderContent').style.display='none';
    return;
  }
  document.getElementById('orderNoDataMsg').style.display='none';
  document.getElementById('orderContent').style.display='block';
  if(!branchOrderRequests[br]){
    branchOrderRequests[br]={date:'—', items:[]};
  }
  var existing={};
  (branchOrderRequests[br].items||[]).forEach(function(it){existing[it._code]=it;});
  branchOrderRequests[br].items=rows.map(function(r){
    var code=gf(r,['كود الصنف']);
    var prev=existing[code];
    // Default suggested qty = ratio - branch stock (gap), min 1
    var gap=Math.max(1,Math.ceil(r._rt-r._cl));
    return {
      _code:code,
      _row:r,
      _reqQty:prev?prev._reqQty:gap,
      _note:prev?prev._note:''
    };
  });
  updateOrderSummary(br);
  renderOrderTable();
}

function updateOrderSummary(br){
  var store = branchOrderRequests[br];
  var items = store ? store.items || [] : [];
  var totalReq = items.reduce(function(s,it){return s+(it._reqQty||0);},0);
  var totalWhStock = getTotalWarehouseStock();
  var whCount = warehouseRows.length;
  var afterReq = totalWhStock - totalReq;

  var elItems = document.getElementById('orderWhTotalItems');
  var elStock = document.getElementById('orderWhTotalStock');
  var elReq = document.getElementById('orderRequestedTotal');
  var elAfter = document.getElementById('orderWhAfterReq');

  if(elItems) elItems.textContent = whCount.toLocaleString();
  if(elStock) elStock.textContent = Math.round(totalWhStock).toLocaleString();
  if(elReq) elReq.textContent = totalReq.toLocaleString();
  if(elAfter){
    elAfter.textContent = Math.max(0,Math.round(afterReq)).toLocaleString();
    elAfter.style.color = afterReq < 0 ? 'var(--rd2)' : 'var(--ac)';
  }
}

function renderOrderTable(){

  var sr=(document.getElementById('orderSrch')?document.getElementById('orderSrch').value.trim():'').toLowerCase();
  var filtered=items.filter(function(it){
    if(rf!=='all'&&it._row._rec!==rf) return false;
    if(sr&&[gf(it._row,['كود الصنف']),gf(it._row,['اسم الصنف']),gf(it._row,['باركود'])].every(function(v){return String(v).toLowerCase().indexOf(sr)===-1;})) return false;
    return true;
  });
  document.getElementById('orderCntLbl').textContent=filtered.length+' صنف';
  var invMap={'خطا كارته':'br','غير متوفر':'bgr','كسر':'ba','متوفر':'bg','اوفر استوك':'bpu'};
  var decMap={'اطلب الان':'order-dec-urgent','قرارات تحسين':'order-dec-improve','مراجعة الصنف':'order-dec-review','متوسط':'order-dec-medium','منتج غير متاح في الفرع':'order-dec-na'};

    return;
  }
  tb.innerHTML=filtered.map(function(it,idx){

    var ib=invMap[r._invStatus]||'bgr';
    var dc=decMap[r._rec]||'order-dec-na';

    var avgDisp=r._avg.toFixed(3);

      +'<td style="color:var(--tx3);font-size:11px">'+gf(r,['باركود'])+'</td>'

      +'<td style="background:rgba(21,128,61,.05);font-weight:700;color:var(--tx)">'+Math.round(r._cl)+'</td>'
      +'<td style="background:rgba(21,128,61,.05);font-weight:600">'+Math.round(r._sa)+'</td>'
      +'<td style="background:rgba(21,128,61,.05)">'+avgDisp+'</td>'
      +'<td style="background:rgba(21,128,61,.05)">'+covDisp+'</td>'
      +'<td style="background:rgba(21,128,61,.05);font-weight:700;color:var(--ac2)">'+r._rp.toFixed(2)+'%</td>'
      +'<td style="background:rgba(21,128,61,.05)">'+(r._transferIndex>=999990?'∞':r._transferIndex.toFixed(3))+'</td>'
      +'<td style="background:rgba(21,128,61,.05)"><span class="badge '+ib+'">'+r._invStatus+'</span></td>'
      +'<td style="background:rgba(21,128,61,.05)"><span class="badge '+dc+'">'+r._rec+'</span></td>'

function handleOrderUpload(e){
  var file=e.target.files[0];if(!file)return;
  if(!user||user.role!=='branch'){e.target.value='';return;}
  var br=user.branch;
  showLoading('جارٍ قراءة ملف الطلبية…','');
  var reader=new FileReader();
  reader.onload=function(ev){
    setTimeout(function(){
      try{
        var wb=XLSX.read(ev.target.result,{type:'array'});
        var ws=wb.Sheets[wb.SheetNames[0]];
        var json=XLSX.utils.sheet_to_json(ws,{defval:''});
        if(!branchOrderRequests[br]) branchOrderRequests[br]={date:'—',items:[]};
        var store=branchOrderRequests[br];
        json.forEach(function(row){
          var code=String(row['كود الصنف']||'').trim();
          var it=store.items.find(function(x){return x._code===code;});
          if(it){it._reqQty=parseInt(row['الكمية المطلوبة'])||0;it._note=String(row['ملاحظات الفرع']||'').trim();}
        });
        renderOrderTable();
        hideLoading();
        alert('✅ تم رفع الطلبية! يمكنك مراجعتها وإرسالها للإدارة.');
      }catch(err){hideLoading();alert('❌ خطأ في قراءة الملف:\n'+err.message);}
    },30);
  };
  reader.readAsArrayBuffer(file);
  e.target.value='';
}

// ══ APPROVE ORDER & DEDUCT WAREHOUSE STOCK ══
function approveOrderForBranch(br){
  var store=branchOrderRequests[br];
  if(!store||!store.items||!store.items.length){alert('لا توجد طلبية.');return;}
  if(store.approved){alert('هذه الطلبية تم اعتمادها مسبقاً وخُصم رصيد المخزن.');return;}
  if(!confirm('اعتماد طلبية فرع '+br+' وخصم الكميات من رصيد المخزن؟')) return;

  var deducted=0, notFound=0, insufficient=0;
  store.items.forEach(function(it){
    if(it._reqQty<=0) return;
    var r=it._row;
    var code=gf(r,['كود الصنف']);
    var barcode=gf(r,['باركود']);
    var cur=getWarehouseStock(code,barcode);
    if(cur===null){notFound++;return;}
    var actual=deductWarehouseStock(code,barcode,it._reqQty);
    if(actual<it._reqQty) insufficient++;
    deducted+=actual;
  });
  store.approved=true;
  store.approvedDate=new Date().toLocaleDateString('ar-EG');
  store.approvedBy=user.name;
  store.status='approved';

  // Refresh all cached calculations
  calcCache.clear();
  refreshDataRepresentation();
  renderBranchOrders();
  updateBoSummary();

  var msg='✅ تم اعتماد طلبية '+br+'!\n';
  msg+='إجمالي مخصوم من المخزن: '+deducted+'\n';
  if(notFound) msg+='⚠️ '+notFound+' صنف لم يُجد في المخزن المركزي\n';
  if(insufficient) msg+='⚠️ '+insufficient+' صنف كان الرصيد أقل من المطلوب (خُصم المتاح)';
  alert(msg);
}

// Approve ALL submitted branch orders
function approveAllBranchOrders(){
  var branches=Object.keys(branchOrderRequests).filter(function(br){
    var s=branchOrderRequests[br];
    return s.status==='submitted'&&!s.approved;
  });
  if(!branches.length){alert('لا توجد طلبيات جديدة للاعتماد.');return;}
  if(!confirm('اعتماد '+branches.length+' طلبية وخصم الكميات من رصيد المخزن؟')) return;
  var totalDeducted=0;
  branches.forEach(function(br){
    var store=branchOrderRequests[br];
    store.items.forEach(function(it){
      if(it._reqQty<=0) return;
      var r=it._row;
      var actual=deductWarehouseStock(gf(r,['كود الصنف']),gf(r,['باركود']),it._reqQty);
      totalDeducted+=actual;
    });
    store.approved=true;
    store.approvedDate=new Date().toLocaleDateString('ar-EG');
    store.approvedBy=user.name;
    store.status='approved';
  });
  calcCache.clear();
  refreshDataRepresentation();
  renderBranchOrders();
  updateBoSummary();
  alert('✅ تم اعتماد '+branches.length+' طلبية!\nإجمالي مخصوم من المخزن: '+totalDeducted);
}

function renderBranchOrders(){

  var branches=Object.keys(branchOrderRequests);
  if(!branches.length){alert('لا توجد طلبيات.');return;}
  showLoading('جارٍ التصدير…','');
  setTimeout(function(){

            'مبيعات الفترة':Math.round(r._sa),
            'نسبة الرشيو%':r._rp.toFixed(2),
            'Transfer Index':r._transferIndex>=999990?'∞':r._transferIndex.toFixed(3),
            'حالة المخزون':r._invStatus,'القرار':r._rec,
            'رصيد المخزن المركزي':liveWh!==null?Math.round(liveWh):Math.round(r._wh),
            'الرشيو المستهدف':Math.round(r._rt),
            'الكمية المطلوبة':it._reqQty,'ملاحظات الفرع':it._note||''
          });
        });
      });
      if(!data.length){hideLoading();alert('⚠️ لا توجد أصناف بكمية > 0.');return;}
      var ws=XLSX.utils.json_to_sheet(data);
      ws['!cols']=Object.keys(data[0]||{}).map(function(k){return{wch:Math.max(k.length*1.8,14)};});
      var wb2=XLSX.utils.book_new();
      XLSX.utils.book_append_sheet(wb2,ws,'طلبيات_الفروع');
      XLSX.writeFile(wb2,'طلبيات_الفروع_'+new Date().toLocaleDateString('ar-EG').replace(/\//g,'-')+'.xlsx');
      hideLoading();
    }catch(err){hideLoading();alert('❌ خطأ:\n'+err.message);}
  },50);
}

function renderTransitPagination(){
  var pg=document.getElementById('trPgNums');if(!pg)return;
  var html='',range=7,half=Math.floor(range/2);
  var start=Math.max(1,trCurPage-half);
  var end=Math.min(trTotalPages,start+range-1);
  if(end-start<range-1) start=Math.max(1,end-range+1);
  if(start>1){html+='<button class="pg-btn" onclick="goTransitPage(1)">1</button>';if(start>2) html+='<span style="color:var(--tx3);padding:0 4px">…</span>';}
  for(var i=start;i<=end;i++) html+='<button class="pg-btn'+(i===trCurPage?' active':'')+'" onclick="goTransitPage('+i+')">'+i+'</button>';
  if(end<trTotalPages){if(end<trTotalPages-1) html+='<span style="color:var(--tx3);padding:0 4px">…</span>';html+='<button class="pg-btn" onclick="goTransitPage('+trTotalPages+')">'+trTotalPages+'</button>';}
  pg.innerHTML=html;
  var f=document.getElementById('trPgFirst'),p=document.getElementById('trPgPrev');
  var n=document.getElementById('trPgNext'),l=document.getElementById('trPgLast');
  if(f) f.disabled=(trCurPage===1);if(p) p.disabled=(trCurPage===1);
  if(n) n.disabled=(trCurPage===trTotalPages);if(l) l.disabled=(trCurPage===trTotalPages);
}

function buildOutboundAction(t){
  if(t.status==='pending_approval') return '<span class="badge status-pending-approval"><i class="fa-solid fa-clock"></i> بانتظار الإدارة</span>';
  if(t.status==='approved') return '<button class="btn bp" style="height:30px;font-size:11px;background:var(--ac2);border-color:var(--ac2)" onclick="executeBranchOutboundCommit(\''+t.id+'\')"><i class="fa-solid fa-truck-ramp-box"></i> تأكيد الشحن</button>';
  if(t.status==='in_transit') return '<span class="badge status-in-transit"><i class="fa-solid fa-truck-moving"></i> في الطريق</span>';
  if(t.status==='rejected') return '<span class="badge status-rejected"><i class="fa-solid fa-ban"></i> مرفوض'+(t.rejectedReason&&t.rejectedReason!=='—'?' — '+t.rejectedReason:'')+'</span>';
  return getStatusBadge(t.status);
}

function buildInboundAction(t){
  if(t.status==='pending_approval') return '<span class="badge status-pending-approval"><i class="fa-solid fa-clock"></i> بانتظار الإدارة</span>';
  if(t.status==='approved') return '<span class="badge status-approved"><i class="fa-solid fa-hourglass-half"></i> بانتظار الشحن من المرسل</span>';
  if(t.status==='in_transit') return '<button class="btn bp" style="height:30px;font-size:11px" onclick="executeBranchInboundCommit(\''+t.id+'\')"><i class="fa-solid fa-square-check"></i> تأكيد الاستلام</button>';

}

// ══ AUTO TRANSIT ══════════════════════════════════════════
function generateAutoTransit(){
  if(!masterBranchRows.length){alert('يرجى رفع ملف الفروع أولاً.');return;}
  var tempTransit=[],itemsByBarcode={};
  var allCalc=masterBranchRows.map(function(r,i){return calcRow(r,i);});
  allCalc.forEach(function(r){
    var bc=String(gf(r,['باركود'])).trim();
    if(!bc||bc===''||bc==='—') return;
    if(!itemsByBarcode[bc]) itemsByBarcode[bc]=[];
    itemsByBarcode[bc].push(Object.assign({},r));
  });
  var matchedCount=0;
  Object.keys(itemsByBarcode).forEach(function(bc){
    var list=itemsByBarcode[bc];
    var needs=list.filter(function(r){return r._transferIndex>cfg.idx_need&&r._hasSales==='يوجد بيع';});
    needs.sort(function(a,b){return b._transferIndex-a._transferIndex;});
    var surplus=list.filter(function(r){return r._transferIndex<cfg.idx_surplus;});
    surplus.sort(function(a,b){
      var aZ=(a._transferIndex===0),bZ=(b._transferIndex===0);
      if(aZ&&bZ) return b._cl-a._cl;
      if(aZ) return -1;if(bZ) return 1;
      return a._transferIndex-b._transferIndex;
    });
    if(!needs.length||!surplus.length) return;
    var ni=0,di=0;
    while(ni<needs.length&&di<surplus.length){
      var need=needs[ni],donor=surplus[di];
      if(donor._branch===need._branch){di++;continue;}
      var needed=Math.max(0,Math.ceil(need._rt-need._cl));
      if(needed<cfg.min_qty){ni++;continue;}
      var allocate;
      if(Math.floor(donor._cl)-needed<Math.ceil(donor._rt)){allocate=Math.floor(donor._cl);}
      else{allocate=needed;}
      if(allocate<cfg.min_qty){di++;continue;}
      var priority=need._transferIndex>=999990?'حرج':(need._transferIndex>cfg.idx_need*3?'عاجل':'عادي');
      var tiNeedDisp=need._transferIndex>=999990?'∞':need._transferIndex.toFixed(4);
      var tiDonorDisp=donor._transferIndex===0?'0 (بدون مبيعات)':donor._transferIndex.toFixed(4);
      var breakNote=Math.floor(donor._cl)-needed<Math.ceil(donor._rt)?' [يتكسر المرسل → كل الرصيد]':'';
      tempTransit.push({
        id:'AUTO-'+(++tc),from:donor._branch,to:need._branch,
        code:gf(need,['كود الصنف']),itemName:gf(need,['اسم الصنف']),barcode:bc,
        qty:allocate,actualQtySent:null,actualQtyReceived:null,
        reason:'TI مرسل: '+tiDonorDisp+' → TI مستلم: '+tiNeedDisp+breakNote,
        priority:priority,status:'pending_approval',isAuto:true,
        date:new Date().toLocaleDateString('ar-EG'),
        approvedBy:null,approvedDate:null,sentDate:null,receivedDate:null,rejectedReason:null,
        fromStock:Math.round(donor._cl),fromSales:Math.round(donor._sa),
        toStock:Math.round(need._cl),toSales:Math.round(need._sa)
      });
      ni++;di++;matchedCount++;
    }
  });
  if(!matchedCount){
    alert('لم يتم العثور على مقترحات ترانزيت.\n\nتلميح: تحقق من إعدادات حدود TI في صفحة الإعدادات.');
  } else {
    transit=tempTransit.concat(transit);
    renderTransit();refreshDataRepresentation();buildSB();
    alert('⚡ تم توليد '+matchedCount+' مقترح ترانزيت!');
  }
}

function executeBranchOutboundCommit(id){
  var t=transit.find(function(x){return x.id===id;});
  if(!t||t.status!=='approved'){alert('الطلب لم يُعتمد.');return;}
  var qty=sn(t.qty);
  var donorItem=masterBranchRows.find(function(row){return row._branch===t.from&&(row['باركود']===t.barcode||row['كود الصنف']===t.code);});
  if(donorItem){var cs=sn(donorItem['رصيد آخر المدة']);if(cs<qty){alert('⚠️ الرصيد ('+cs+') أقل من المطلوب ('+qty+')!');return;}}
  var actualStr=prompt('الكمية الفعلية المشحونة (المطلوبة: '+qty+'):',String(qty));
  if(actualStr===null) return;
  var actualQty=parseInt(actualStr)||0;
  if(actualQty<=0){alert('الكمية يجب أن تكون أكبر من صفر.');return;}
  if(donorItem){var cs2=sn(donorItem['رصيد آخر المدة']);if(cs2<actualQty){alert('⚠️ الكمية تتجاوز الرصيد!');return;}donorItem['رصيد آخر المدة']=cs2-actualQty;}
  t.status='in_transit';t.actualQtySent=actualQty;t.sentDate=new Date().toLocaleDateString('ar-EG');
  var idx=masterBranchRows.indexOf(donorItem);
  if(idx>-1){calcCache.forEach(function(v,k){if(k.startsWith(idx+'|')) calcCache.delete(k);});}
  alert('📦 تم تأكيد الشحن! الكمية: '+actualQty);
  refreshDataRepresentation();buildSB();renderTransit();
}

function executeBranchInboundCommit(id){
  var t=transit.find(function(x){return x.id===id;});
  if(!t||t.status!=='in_transit'){alert('الشحنة ليست في الطريق.');return;}
  var sentQty=t.actualQtySent||t.qty;
  var actualStr=prompt('الكمية الفعلية المستلمة (المرسلة: '+sentQty+'):',String(sentQty));
  if(actualStr===null) return;
  var actualQty=parseInt(actualStr)||0;
  if(actualQty<=0){alert('الكمية يجب أن تكون أكبر من صفر.');return;}
  var receiverItem=masterBranchRows.find(function(row){return row._branch===t.to&&(row['باركود']===t.barcode||row['كود الصنف']===t.code);});
  if(receiverItem){
    receiverItem['رصيد آخر المدة']=sn(receiverItem['رصيد آخر المدة'])+actualQty;
    var idx=masterBranchRows.indexOf(receiverItem);
    if(idx>-1){calcCache.forEach(function(v,k){if(k.startsWith(idx+'|')) calcCache.delete(k);});}
  }
  t.status='received';t.actualQtyReceived=actualQty;t.receivedDate=new Date().toLocaleDateString('ar-EG');
  var diff=actualQty!==sentQty?'\n⚠️ فرق: مرسل '+sentQty+' — مستلم '+actualQty:'';
  alert('🎉 تم الاستلام! الكمية: '+actualQty+diff);
  refreshDataRepresentation();buildSB();renderTransit();
}
